Plot ideograms

chromsizes<-read.table("/home/l.brueckner/mydata/annotations/chromsizes_mm10.txt",header=F, stringsAsFactors = F)
names(chromsizes)<-c("chr_r","pos_r")
chromsizes<-chromsizes[chromsizes$chr_r %in% paste0("chr",c(1:19,"X")),]
chromsizes_filtered<-subset(chromsizes, chr_r %in% paste0("chr",c(1:19,"X")))
chromsizes_filtered$chr_r<-factor(chromsizes_filtered$chr_r, paste0("chr",c(1:19,"X"))) #sort chr by human logic


LADint_plot<-IntE[IntE$LADstate=="LAD" & !is.na(IntE$LADstate), c("seqnames","start")]
iLADint_plot<-IntE[IntE$LADstate=="iLAD" & !is.na(IntE$LADstate), c("seqnames","start")]
names(LADint_plot)<-c("chr_r", "pos_r")
names(iLADint_plot)<-c("chr_r", "pos_r")


PPLOT<-ggplot(chromsizes_filtered, aes(x=chr_r,y=as.numeric(pos_r)))+
  theme_minimal()+
  geom_bar(stat='identity',fill="#dddddd",color="#dddddd")+
  geom_hline(data = LADint_plot, aes(yintercept= as.numeric(pos_r),color="blue"))+
  geom_hline(data = iLADint_plot, aes(yintercept= as.numeric(pos_r),color="red"))+
  facet_grid(. ~ chr_r, scale = "free", space = "free_x")+
  scale_y_continuous(breaks = c(0, 50000000,100000000,150000000,200000000), labels=c("0", "50","100","150","200 Mb"), ) +
  scale_color_manual(labels = c( "LAD","iLAD"), values = c("darkorange","#222222bb")) +
  guides(color=guide_legend(""))+
  theme(axis.title.y = element_blank(), axis.ticks.x = element_blank(),axis.text.y = element_text(angle=90, size = 14),axis.title.x = element_blank(), axis.text.x = element_blank(),strip.text.x = element_text(angle= 90, size=16))

PPLOT
