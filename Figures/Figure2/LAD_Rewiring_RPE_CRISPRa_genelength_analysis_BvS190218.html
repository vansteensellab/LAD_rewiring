LAD rewiring data: CRISPRa in RPE cells – link between gene length and detached region size.

Bas van Steensel, 23-28 Dec 2018, 18 Feb 2019.

Introduction

Plot domainograms around all targeted genes in a compact manner. Aim is to visualize the relationship between length of the targeted genes and the size of the detached region. For this purpose all domainograms are shown for the same size of genomic DNA (~2Mb).

Version history

Description of Data

  • Domainograms were generated in scripts based on LAD_Rewiring_RPE_CRISPRa_template_BvS181223.Rhtml

#slightly modified domainogram plot function (DOES NOT PLOT QUANTILES AT RED/LOW END)
plot.domainogram.nored<-function(dg, start=1, end=Inf, x.axis=TRUE, ...)
  #input: 
  #dg: GRanges object of a single chromosome, with domainogram data as metadata, generated by domainogram.bvs() function
  #start: start chromosomal coordinate of plot (in bp)
  #end: end chromosomal coordindate of plot (in bp)
  #... par parameters to be passed on to plot function
{chrom<-unique(seqnames(dg))
stopifnot(length(chrom)==1)
color.fun1 <- colorRampPalette(c("white","white"))
color.fun2 <- colorRampPalette(c("white", "navyblue"))
color.vec <- c(color.fun1(51),rep("#FFFFFF",899),color.fun2(51))
if(end == Inf){end<-max(end(dg))}
xslack<-(end-start)*0.04 #same as regular plot function: add slack on each side on the x-axis
dgsub<-dg[start(dg)>=start-xslack & end(dg)<=end+xslack] #reduce dg to plotting region
xpos<-round((start(dgsub)+end(dgsub))/2)
image(x=xpos, y=c(1:ncol(mcols(dgsub))), z=as.matrix(mcols(dgsub)), col=color.vec,
      yaxt="n", xaxt="n", ...)
box()
#x-axis:
if(x.axis==TRUE){axis(side=1, axTicks(1), labels=round(axTicks(1)/1e6,1))}
#y-axis:
gatc2len<-mean(width(dgsub)) #average gatc fragment length
wsizes<-log10(as.integer(names(mcols(dgsub)))*gatc2len)
at<-pretty(wsizes, n=3) #log10 window sizes tick marks
#calculation y-axis coordinates of these tick marks
fullrange<-range(wsizes)
tpos<-((at-min(fullrange))*length(wsizes)/diff(fullrange))+0.5
axis(side=2, at=tpos, labels=at, ...)
}

Load, align and plot domainograms

plot.aligned.domainograms<-function(Gdf, sgRNA){
#sort genes by distance from sgRNA to gene end:
row.names(Gdf)<-Gdf$gene
print(Gdf)
Gdf<-Gdf[row.names(sgRNA),]
print(Gdf)
Gdf<-Gdf[order(abs(ifelse(Gdf$strand=="+",Gdf$end,Gdf$start)-sgRNA$start)),]
#plot all domainograms, blue only:
par(mar=c(1,0,0,0), mgp=c(1.8,0.5,0), oma=c(2,2,0.5,0.5), lend=1)
par(mfrow=c(length(genes),1))
for(G in as.character(Gdf$gene))
{print(G)
 Gcenter<-sgRNA[G,"start"]
 grange<-as.numeric(Gdf[G,c("start","end")])
 #load domainogram:
 load(file=paste0("EmpiricalDomainogram_chrExp_LAD_rewiring_RPE_", G,"_BvS_190218.Rdata"))
 DG.sub<-DG.Exp[start(DG.Exp)>=Gcenter-ifelse(Gdf[G,"strand"]=="+",PlotRangeU,PlotRangeD) & start(DG.Exp)<=Gcenter+ifelse(Gdf[G,"strand"]=="+",PlotRangeD,PlotRangeU)]
 #flip data if gene on - strand:
 if(Gdf[G, "strand"]=="-")
 {#invert coordinates by subtracting them from the largest coordinate
  DG.sub2<-DG.sub
  start(DG.sub2)<-rev(max(end(DG.sub))-end(DG.sub)+1)
  width(DG.sub2)<-rev(width(DG.sub))
  #also reverse domainogram data matrix:
  TT<-as.matrix(mcols(DG.sub))
  mcols(DG.sub2)<-TT[nrow(TT):1,]
  #adjust sgRNA and gene coordinates:
  Gcenter<-max(end(DG.sub)-Gcenter+1)
  grange[1]<-max(end(DG.sub)-grange[1]+1)
  grange[2]<-max(end(DG.sub)-grange[2]+1)
  DG.sub<-DG.sub2
  }
 plot.domainogram.nored(DG.sub, x.axis=FALSE, xlab="", ylab="")
 arrows(x0=grange[1], y0=0, x1=grange[2], lwd=2.5, col="#55BB2F", code=ifelse(Gdf[G,'strand']=="+",2, 1), angle=60, length=0.07, xpd=NA)
 legend("topleft", G, bty="n", text.col="#220000")
 abline(v=Gcenter, lty="dotted", col="#55992F", lwd=3)
}
#add y-axis
tstart<-par("usr")[1]
ticks<-pretty(c(-PlotRangeU,PlotRangeD))
axis(side=1, at=tstart+ticks+PlotRangeU-100, labels=round(ticks/1e6,1), lwd=0, lwd.ticks=1)
mtext("position relative to sgRNA target [Mb]", side=1, line=1.8)
}


#plot:
plot.aligned.domainograms(Gdf=Gdf, sgRNA=sgRNA)
